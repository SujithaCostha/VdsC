//LOAD
int load(int value)
{
  int amount;
  printf("\t\t\tEnter the loading amount : ");
  scanf("%d",&amount);

  updateStock(-amount);
  value+=amount;


  return value;
}

//UNLOAD
int unload(int value)
{
  int amount;
  printf("\t\t\tEnter the unloading amount : ");
  scanf("%d",&amount);

  updateStock(amount);
  value-=amount;


  return value;

}

//STOCK UPDATES
int updateStock(int val)
{
    FILE *stock;
    stock = fopen("Stock.txt", "w+");
    fscanf(stock,"%d", &stockBalance);
    stockBalance+=val;
    fprintf(stock, "%d", stockBalance);
    fclose(stock);
    return stockBalance;
}
